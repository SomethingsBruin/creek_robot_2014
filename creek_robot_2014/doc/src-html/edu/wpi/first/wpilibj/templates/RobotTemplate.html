018/**
019 * The VM is configured to automatically run this class, and to call the
020 * functions corresponding to each mode, as described in the IterativeRobot
021 * documentation. If you change the name of this class or the package after
022 * creating this project, you must also update the manifest file in the resource
023 * directory.
024 */
025public class RobotTemplate extends IterativeRobot 
026{
027    //The robot driver.
028    private Driver _driver;
029    
030    //The robot chassis.
031    private Chassis _chassis;
032    
033    //The robot mechanism.
034    private Mechanism _mechanism;
035    
036    //The AutoCommand to be run in autonomous.
037    private AutoCommand _autoCommand;
038    
039    //Declares the Smart Dashboard device which chooses the Driver and the drive type.
040    private SendableChooser _driverChooser;
041    private SendableChooser _driveTypeChooser;
042    
043    //Declares the Smart Dashboard device which chooses which AutoCommand to run.
044    private SendableChooser _autoCommandChooser;
045    
046    //A flag that insure autonomous only goes once.
047    private boolean _autoFlag = true;
048
049    /**
050     * This function is run when the robot is first started up and initializes
051     * the mechanism and chassis, and does all the Smart Dashboard initialization.
052     */
053    public void robotInit() 
054    {
055        //Get the chassis object.
056        _chassis = Chassis.getInstance();
057        
058        //Get the mechanism object.
059        _mechanism = Mechanism.getInstance();
060        
061        //Initializes the chooser devices.
062        _driverChooser = new SendableChooser();
063        _driveTypeChooser = new SendableChooser();
064        _autoCommandChooser = new SendableChooser();
065        
066        //Assigns a index number to each Driver type. The Airplane Controller is the default selection.
067        _driverChooser.addDefault( "Airplane Controller" , new Integer( 0 ) );//0 for the Airplane Controller.
068        _driverChooser.addObject( "Attack Three" , new Integer( 1 ) );//1 for the Attack Three joysticks.
069        _driverChooser.addObject( "XBox Controller" , new Integer( 2 ) );//2 for the XBox Controller.
070        
071        //Assigns a index number to each drive type. The Relative Holo Drive is the default selection.
072        _driveTypeChooser.addDefault( "Relative Holo Drive" , new Integer( 0 ) );//0 for the Relative Holo Drive.
073        _driveTypeChooser.addObject( "Normal Holo Driver" , new Integer( 1 ) );//1 for the Normal Holo Dirve.
074        
075        //Adds each AutoCommand into the Smart Dashboard.
076        _autoCommandChooser.addDefault( "Do Nothing" , new AutoNothing() );
077        _autoCommandChooser.addObject( "Center Auto Command" , new AutoCenter() );
078        _autoCommandChooser.addObject( "Side Auto Command" , new AutoSide() );
079        
080        //Puts the chooser devices into the Smart Dashboard.
081        SmartDashboard.putData( "Driver Controller" , _driverChooser );
082        SmartDashboard.putData( "Drive Type" , _driveTypeChooser );
083        SmartDashboard.putData( "Auto Command" , _autoCommandChooser );
084    }
085    
086    /**
087     * This function is called once when robot is disabled and prompts the user
088     * that the robot is disabled. Also resets the gyro on the robot to 0 degrees.
089     */ 
090    public void disabledInit()
091    {
092        //Prompts that the robot is disabled.
093        System.out.println( "Robot is Disabled" );
094        
095        //Sets the automous flag to be false.
096        _autoFlag =  false;
097        
098        //Resets the gyro to 0 degrees.
099        _chassis.resetGyro();
100    }
101    
102    /**
103     * A function which is called once at the beginning of Autonomous and finds which
104     * AutoCommand that will be run, resets the gyro, and cocks the mechanism.
105     */
106    public void autonomousInit()
107    {
108        //Finds the selected AutoCommand.
109        _autoCommand = (AutoCommand) _autoCommandChooser.getSelected(); 
110        
111        //Resets the gyro
112        _chassis.resetGyro();
113        
114        //Initially cocks the mechanism shooter.
115        _mechanism.shoot();
116    }
117
118    /**
119     * This function is called periodically during autonomous and runs the given
120     * AutoCommand once.
121     */
122    public void autonomousPeriodic() 
123    {
124        //If the flag hasn't been raised...
125        if( !_autoFlag )
126        {
127            //Runs the given AutoCommand and set the auto flag to be true.
128            _autoCommand.runAutoCommand();
129            _autoFlag = true;
130        }
131    }
132    
133    /**
134     * A function which is called once at the beginning of Tele-Op and finds which
135     * driver type is wanted, resets the gyro, and cocks the mechanism.
136     */
137    public void teleopInit()
138    {
139        //Finds the assigned index value of the driver type choosen
140        int index = ( (Integer) _driverChooser.getSelected() ).intValue();
141        
142        //The type of the driver will be choosen from the given index value from the Smart Dashboard.
143        switch( index )
144        {
145            //The XBox Controller if the index is 2.
146            case 2:
147                _driver = XBoxController.getInstance();
148                break;
149                
150            //The Attack Three joysticks if the index is 1.
151            case 1:
152                _driver = AttackThree.getInstance();
153                break;
154            
155            //The Airplane Controller if the index is 0 (or anything else).
156            default:
157            case 0:
158                _driver = AirplaneController.getInstance();
159                break;
160        }
161        
162        //Reset the gyro.
163        _chassis.resetGyro();
164        
165        //Initially cocks the mechanism.
166        _mechanism.shoot();
167    }
168
169    /**
170     * This function is called periodically during operator control and it will
171     * drive the robot relative to the driver, not its starting position. The robot
172     * will square itself to the wall on command.
173     */
174    public void teleopPeriodic() 
175    {
176        //Finds which drive type is wanted from the SmartDashBoard.
177        int index = ( (Integer) _driveTypeChooser.getSelected() ).intValue();
178        
179        //Based on the selection above, choose and run the selected drive type.
180        switch( index )
181        {
182            //1 is Normal Holo Drive.
183            case 1:
184                //Drives the chassis not relative to the driver.
185                _chassis.holoDrive( _driver.getY() , _driver.getX() , _driver.getRot() );
186                break;
187            
188            //0 and default is Relative Holo Drive.
189            default:
190            case 0:
191                //Drives the chassis relative to the driver.
192                _chassis.relativeHoloDrive( _driver.getY() , _driver.getX() , _driver.getRot() );
193                break;
194        }      
195        
196        //If the primary button is pressed...
197        if( _driver.getPriButton() )
198        {
199            //Square the robot back to the wall.
200            _chassis.square( 0.8 );
201        }
202        
203        //If the secondary button is pressed...
204        if( _driver.getSecButton() )
205        {
206            //Shoot the mechanism.
207            _mechanism.shoot();
208        }
209        
210        //If the third button is pressed...
211        if( _driver.getThirdButton() )
212        {
213            //The arm on the mechanism will raise at 0.8 speed.
214            _mechanism.raiseArm( 0.8 );
215        }
216        else if( _driver.getFourthButton() )//Else if the fourth button is pressed...
217        {
218            //The arm on the mechanism will lower at 0.8 speed.
219            _mechanism.lowerArm( 0.8 );
220        }
221        
222        //If the fifth button is pressed...
223        if( _driver.getFifthButton() )
224        {
225            //The mechanism will eject the ball.
226            _mechanism.eject();
227        }
228        else if( _driver.getSixthButton() )//Else if the sixth button is pressed...
229        {
230            //The mechanism will intake the ball.
231            _mechanism.intake();
232        }
233        else//Else the intake motors on the mechanism will stop.
234        {
235            _mechanism.stopIntake();
236        }
237    }
