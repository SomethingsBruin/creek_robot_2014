019/**
020 * The VM is configured to automatically run this class, and to call the

026public class RobotTemplate extends IterativeRobot 
027{
028    //The robot driver.
029    private Driver _driver;
030    
031    //The robot chassis.
032    private Chassis _chassis;
033    
034    //Initializes the original PID constants for the chassis. These are dynamically changable in the Smart Dashboard.
035    private final double _KP = 0.7;
036    private final double _KI = 0.008;
037    private final double _KD = 0.04;
038    
039    //The robot mechanism.
040    private Mechanism _mechanism;
041    
042    //The maximum and minimum extremes for the position of the arm
043    private final double _MAX_ARM_EXTREME = 450.0;
044    private final double _MIN_ARM_EXTREME = 980.0;
045    
046    //The maximum up and down speed of the arm.
047    private final double _ARM_UP_SPEED = 0.7;
048    private final double _ARM_DOWN_SPEED = -0.6;
049          
050    //The delay between ejecting the ball and shooting the ball.
051    private final double _DELAY = 0.1;
052    
053    //The AutoCommand to be run in autonomous.
054    private AutoCommand _autoCommand;
055    
056    //Declares the Smart Dashboard device which chooses the Driver and the drive type.
057    private SendableChooser _driverChooser;
058    private SendableChooser _driveTypeChooser;
059    
060    //Declares the Smart Dashboard device which chooses which AutoCommand to run.
061    private SendableChooser _autoCommandChooser;
062    
063    //A flag that insure autonomous only goes once.
064    private boolean _autoFlag = true;
065
066    /**
067     * This function is run when the robot is first started up and initializes
068     * the mechanism and chassis, and does all the Smart Dashboard initialization.
069     */
070    public void robotInit() 
071    {
072        //Get the chassis object.
073        _chassis = Chassis.getInstance();
074        
075        //Puts the PID constants into the Smart Dashboard so they are dynamicly changable.
076        SmartDashboard.putNumber( " P-Constant: ", _KP );
077        SmartDashboard.putNumber( " I-Constant: ", _KI );
078        SmartDashboard.putNumber( " D-Constant: ", _KD );
079        
080        //Get the mechanism object.
081        _mechanism = Mechanism.getInstance();
082        
083        //Puts the maximum arm up and down speeds into the SmartDashboard
084        SmartDashboard.putNumber( " Arm Up-Speed: " , _ARM_UP_SPEED );
085        SmartDashboard.putNumber( " Arm Down-Speed: " , _ARM_DOWN_SPEED );
086        
087        //Puts the minimum and maximum arm extremes into the SmartDashboard.
088        SmartDashboard.putNumber( " Arm Maximum Extreme: " , _MAX_ARM_EXTREME );
089        SmartDashboard.putNumber( " Arm Minimum Extreme: " , _MIN_ARM_EXTREME );
090        
091        //Puts the arm shooting delay into the SmartDashboard.
092        SmartDashboard.putNumber( " Arm Shooting Delay: " , _DELAY );
093        
094        //Initializes the chooser devices.
095        _driverChooser = new SendableChooser();
096        _driveTypeChooser = new SendableChooser();
097        _autoCommandChooser = new SendableChooser();
098        
099        //Assigns a index number to each Driver type. The Airplane Controller is the default selection.
100        _driverChooser.addDefault( "Airplane Controller" , new Integer( 0 ) );//0 for the Airplane Controller.
101        _driverChooser.addObject( "Attack Three" , new Integer( 1 ) );//1 for the Attack Three joysticks.
102        _driverChooser.addObject( "XBox Controller" , new Integer( 2 ) );//2 for the XBox Controller.
103        
104        //Assigns a index number to each drive type. The Normal Holo Drive is the default selection.
105        _driveTypeChooser.addDefault( "Normal Holo Driver" , new Integer( 1 ) );//1 for the Normal Holo Dirve.
106        _driveTypeChooser.addObject( "Relative Holo Drive" , new Integer( 0 ) );//0 for the Relative Holo Drive.
107        
108        //Adds each AutoCommand into the Smart Dashboard.
109        _autoCommandChooser.addDefault( "Do Nothing" , new AutoNothing() );
110        _autoCommandChooser.addObject( "Center Auto Command" , new AutoCenter() );
111        _autoCommandChooser.addObject( "Side Auto Command" , new AutoSide() );
112        
113        //Puts the chooser devices into the Smart Dashboard.
114        SmartDashboard.putData( "Driver Controller" , _driverChooser );
115        SmartDashboard.putData( "Drive Type" , _driveTypeChooser );
116        SmartDashboard.putData( "Auto Command" , _autoCommandChooser );
117        
118        //Finds the assigned index value of the driver type choosen
119        int index = ( (Integer) _driverChooser.getSelected() ).intValue();
120        
121        //The type of the driver will be choosen from the given index value from the Smart Dashboard.
122        switch( index )
123        {
124            //The XBox Controller if the index is 2.
125            case 2:
126                _driver = XBoxController.getInstance();
127                break;
128                
129            //The Attack Three joysticks if the index is 1.
130            case 1:
131                _driver = AttackThree.getInstance();
132                break;
133            
134            //The Airplane Controller if the index is 0 (or anything else).
135            default:
136            case 0:
137                _driver = AirplaneController.getInstance();
138                break;
139        }
140    }

144     * that the robot is disabled. Also resets the gyro on the robot to 0 degrees.
145     */

151        //Sets the automous flag to be false.
152        _autoFlag = false;
153

158    /**
159     * A function which is called once at the beginning of Autonomous and finds which
160     * AutoCommand that will be run, resets the gyro, and cocks the mechanism.
161     */
162    public void autonomousInit()
163    {
164        //Finds the selected AutoCommand.
165        _autoCommand = (AutoCommand) _autoCommandChooser.getSelected(); 
166        
167        //Resets the gyro
168        _chassis.resetGyro();
169    }

171    /**
172     * This function is called periodically during autonomous and runs the given
173     * AutoCommand once.
174     */
175    public void autonomousPeriodic() 
176    {
177        //If the flag hasn't been raised...
178        if( !_autoFlag )
179        {
180            //Runs the given AutoCommand and set the auto flag to be true.
181            _autoCommand.runAutoCommand();
182            _autoFlag = true;
183        }
184    }

186    /**
187     * A function which is called once at the beginning of Tele-Op and finds which
188     * driver type is wanted, resets the gyro, and cocks the mechanism.
189     */
190    public void teleopInit()
191    {
192        //Finds the assigned index value of the driver type choosen
193        int index = ( (Integer) _driverChooser.getSelected() ).intValue();
194        
195        //The type of the driver will be choosen from the given index value from the Smart Dashboard.
196        switch( index )
197        {
198            //The XBox Controller if the index is 2.
199            case 2:
200                _driver = XBoxController.getInstance();
201                break;
202                
203            //The Attack Three joysticks if the index is 1.
204            case 1:
205                _driver = AttackThree.getInstance();
206                break;
207            
208            //The Airplane Controller if the index is 0 (or anything else).
209            default:
210            case 0:
211                _driver = AirplaneController.getInstance();
212                break;
213        }
214        
215        //Reset the gyro.
216        _chassis.resetGyro();
217    }

219    /**
220     * This function is called periodically during operator control and it will
221     * drive the robot relative to the driver, not its starting position. The robot
222     * will square itself to the wall on command.
223     */
224    public void teleopPeriodic() 
225    {
226        //Finds which drive type is wanted from the SmartDashBoard.
227        int index = ( (Integer) _driveTypeChooser.getSelected() ).intValue();
228        
229        //Based on the selection above, choose and run the selected drive type.
230        switch( index )
231        {
232            //1 is Normal Holo Drive.
233            case 1:
234                //Drives the chassis not relative to the driver.
235                _chassis.holoDrive( _driver.getY() , _driver.getX() , _driver.getRot() );
236                break;
237            
238            //0 and default is Relative Holo Drive.
239            default:
240            case 0:
241                //Drives the chassis relative to the driver.
242                _chassis.relativeHoloDrive( _driver.getY() , _driver.getX() , Utility.limitRange( _driver.getRot(), 0.75, -0.75 ) );
243                break;
244        }
245        
246        //If the analog button's sum is negative and the arm is below the minimum extreme...
247        if( !_mechanism.isSettingArm() && _driver.getArm() < 0.0 && _mechanism.getPotentValue() < SmartDashboard.getNumber( " Arm Minimum Extreme: " ) )
248        {
249            //The arm on the mechanism will raise at the analog speed.
250            _mechanism.lowerArm( _driver.getArm() );
251        }
252        else if( !_mechanism.isSettingArm() && _driver.getArm() > 0.0 && _mechanism.getPotentValue() > SmartDashboard.getNumber( " Arm Maximum Extreme: " ) )//Else if the analog button's sum is positive and the arm is above the minimum extreme...
253        {
254            //The arm on the mechanism will lower at analog speed.
255            _mechanism.raiseArm( -1 * _driver.getArm() );
256        }
257        else//Else stop the arm.
258        {
259            _mechanism.stopArm();
260        }
261        
262        //If the primary button is pressed...
263        if( _driver.getPriButton() )
264        {
265            //Square the robot back to the wall.
266            _chassis.square( 0.8 );
267        }
268        
269        //If the secondary button is pressed and the arm isn't being set...
270        if( _driver.getSecButton() && !_mechanism.isSettingArm() )
271        {
272            //Shoot the mechanism.
273            _mechanism.shoot();
274        }
275        
276        //If the arm is currently not being set.
277        if( !_mechanism.isSettingArm() )
278        {
279            //If the third button is pressed...
280            if( _driver.getThirdButton() )
281            {
282                //Set the arm to the bottom state.
283                _mechanism.setArm( 0 );
284            }
285            else if( _driver.getFourthButton() )//If the fourth button is pressed...
286            {
287                //Set the arm to the top state.
288                _mechanism.setArm( 1 );
289            }
290            else if( _driver.getFifthButton() )//If the fifth button is pressed...
291            {
292                //Set the arm to the middle state.
293                _mechanism.setArm( 2 );
294            }
295        }
296        
297        //If the fifth button is pressed...
298        if( _driver.getSixthButton() )
299        {
300            //The mechanism will eject the ball.
301            _mechanism.eject();
302        }
303        else if( _driver.getSeventhButton() )//Else if the sixth button is pressed...
304        {
305            //The mechanism will intake the ball.
306            _mechanism.intake();
307        }
308        else if( !_mechanism.isShooting() )//Else if the mechanism is not shooting...
309        {
310            //The intake motors on the mechanism will stop.
311            _mechanism.stopIntake();
312        }
313    }
