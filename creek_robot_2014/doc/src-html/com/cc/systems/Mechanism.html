001package com.cc.systems;
002
003import com.cc.outputs.motors.CCTalon;
004import com.cc.outputs.motors.CCVictor;
005import com.cc.shooter.*;
006
007import edu.wpi.first.wpilibj.AnalogChannel;
008import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
009
010/**
011 * This class represents the Mechanism on the robot and is responsible for
012 * operating the shooter and the pick-up mechanism of the robot.
013 */
014public class Mechanism 
015{
016    //The singleton object of the Mechanism class.
017    private static Mechanism _instance = null;
018
019    //The talon which represents the pivot of the mechanism arm.
020    //private CCTalon _pivot;
021    private CCVictor _pivot;
022
023    //The victors which represent the motors which intake the balls into the mechanism.
024    private CCVictor _intakeOne;
025    private CCVictor _intakeTwo;
026
027    //The actual shooter mechanism and the thread which cocks the shooter.
028    private Shooter _shooter;
029    private ShooterReset _shooterReset;
030    
031    //The potentiometer object of the mechanism. 
032    private AnalogChannel _potent;
033    
034    //The delay between ejecting the ball and shooting the ball.
035    private final double _DELAY = 0.1;
036
037    private Mechanism() 
038    {
039        //Initializes the pivot talon.
040        //_pivot = new CCTalon( 5, false );
041        _pivot = new CCVictor( 5, false );
042
043        //Initializes the victors for the intake.
044        _intakeOne = new CCVictor( 6, false );
045        _intakeTwo = new CCVictor( 7, false );
046        
047        //Gets singleton of the shooter obejct.
048        _shooter = Shooter.getInstance();
049        
050        //Initializes the potentiometer on channel 1.
051        _potent = new AnalogChannel( 1 );
052        
053        //Puts the arm shooting delay into the smartdashboard.
054        SmartDashboard.putNumber( " Arm Shooting Delay: " , _DELAY );
055    }
056
057    /**
058     * Returns the singleton object of the Mechanism..
059     *
060     * @return The singleton object of the Mechanism.
061     */
062    public static Mechanism getInstance() 
063    {
064        //If the instance of the mechanism has not been initialized yet then...
065        if( _instance == null ) 
066        {
067            //Initialize the singletone Mechanism object.
068            _instance = new Mechanism();
069        }
070
071        //Return the singleton object of the Mechanism.
072        return _instance;
073    }
074
075    /**
076     * Shoots the shooter and then cocks the shooter.
077     */
078    public void shoot() 
079    {
080        //If the thread has not been created...
081        if( _shooterReset == null )
082        {
083            //Create a new thread.
084            _shooterReset = new ShooterReset( _shooter, SmartDashboard.getNumber(  " Arm Shooting Delay: " ) );
085        }
086        
087        //If we are not already shooting...
088        if( !_shooterReset.isAlive() ) 
089        {
090            //Create a new thread and then run that thread.
091            _shooterReset = new ShooterReset( _shooter, SmartDashboard.getNumber( " Arm Shooting Delay: " ) );
092            _shooterReset.start();
093        }
094    }
095    
096    /**
097     * Sets the motors on the mechanism to intake the ball.
098     */
099    public void intake()
100    {
101        //Sets both motors to full forward to intake the ball.
102        _intakeOne.set( 1.0 );
103        _intakeTwo.set( 1.0 );       
104    }
105    
106    /**
107     * Sets the motors on the mechanism to eject the ball.
108     */
109    public void eject()
110    {
111        //Set both motors to full reverse to eject the ball.
112        _intakeOne.set( -1.0 );
113        _intakeTwo.set( -1.0 );      
114    }
115    
116    /**
117     * Stops the intake motors on the mechanism.
118     */
119    public void stopIntake()
120    {
121        //Stops both motors.
122        _intakeOne.set( 0.0 );
123        _intakeTwo.set( 0.0 );       
124    }
125    
126    /**
127     * Lowers the arm on the mechanism at the given speed.
128     * 
129     * @param speed The speed at which to lower the arm.
130     */
131    public void lowerArm( double speed )
132    {
133        //Raises the arm at the given speed.
134        _pivot.set( speed );
135    }
136    
137    /**
138     * Raises the arm on the mechanism at the given speed.
139     * 
140     * @param speed The speed at which to raise the arm.
141     */
142    public void raiseArm( double speed )
143    {
144        //Lowers the arm at the given speed.
145        _pivot.set( -speed );
146    }
147    
148    /**
149     * Stops the arm on the mechanism were it is.
150     */
151    public void stopArm()
152    {
153        _pivot.set( 0.0 );
154    }
155    
156    /**
157     * Returns whether the mechanism is shooting.
158     * 
159     * @return Whether the mechanism is shooting or not.
160     */
161    public boolean isShooting()
162    {
163        //Gets whether the mechanism is shooting and returns it.
164        boolean isAlive = _shooterReset != null && _shooterReset.isAlive();
165        return isAlive;
166    }
167    
168    /**
169     * Gets the current value of the potentiometer on the arm.
170     * 
171     * @return The value of the potentiometer.
172     */
173    public double getPotentValue()
174    {
175        //Finds the potentiometer value and returns it.
176        double value = _potent.getValue();
177        return value;
178    }
179}
