070    /**
071     * Moves the robot using the mecanum wheels given a forward, slide, and
072     * rotation value.
073     * 
074     * @param fwd The forward value from a driver.
075     * @param sld The slide value from a driver.
076     * @param rot The rotation value from a driver.
077     */
078    public void holoDrive( double fwd, double sld, double rot )
079    {
080        //Calcuates the vector for each motor given the forward, slide, and rotation values.
081        double lf = fwd + sld + rot;
082        double rf = fwd - sld - rot;
083        double lr = fwd - sld + rot;
084        double rr = fwd + sld - rot;
085        
086        //Makes the max value equal to the left motor.
087        double maxVal = Math.abs( lf );
088        
089        //If any motor value is greater than the max value, make the max value equal to that motor value.
090        if ( Math.abs( rf ) > maxVal )
091        {
092            maxVal = Math.abs( rf );
093        }
094        
095        if ( Math.abs( lr ) > maxVal )
096        {
097            maxVal = Math.abs( lr );
098        }
099        
100        if ( Math.abs( rr ) > maxVal )
101        {
102            maxVal = Math.abs( rr );
103        }
104        
105        //If the max value was greater than one or less than negative one...
106        if( maxVal > 1.0 )
107        {
108            //Limit that motor value and scale the other motor values appropriatly.
109            lf /= maxVal;
110            rf /= maxVal;
111            lr /= maxVal;
112            rr /= maxVal;
113        }
114         
115        //Set each motor to its corresponding value.
116        _leftFront.set( lf );
117        _rightFront.set( rf );
118        _leftRear.set( lr );
119        _rightRear.set( rr );
120    }

122    /**
123     * Moves the robot based on the mecanum system given a forward value, a slide value,
124     * and a rotation value. Moves relative to driver instead of the starting position. 
125     * 
126     * @param fwd The forward value from a driver.
127     * @param sld The slide value from a driver.
128     * @param rot The rotation value from a driver.
129     */
130    public void relativeHoloDrive( double fwd, double sld, double rot )
131    {
132        //Calculates the relative forward and slide values.
133        double fwdTemp = fwd * Math.cos( getGyro() ) + sld * Math.sin( getGyro() );
134        double sldTemp = sld * Math.cos( getGyro() ) - fwd * Math.sin( getGyro() );
135        
136        //Inputs the calculated values into the normal holoDrive function.
137        holoDrive( fwdTemp, sldTemp, rot );     
138    }

140    /**
141     * Returns the angle of the gyro in radians on the chassis.
142     * 
143     * @return The angle of the gyro in radians.
144     */
145    public double getGyro()
146    {
147        //Gets the raw angle from the gyro
148        double gyroValue = _gyro.getAngle();
149        
150        //If the angle is greater than 360 or less than -360...
151        if ( Math.abs( gyroValue ) >= 360)
152        {
153            //Reset the gyro to zero degrees.
154            _gyro.reset();
155            gyroValue = _gyro.getAngle();
156        }
157        
158        //Converts the angle of the gyro from degrees to radians.
159        gyroValue = Math.toRadians( gyroValue );
160        
161        //Returns the gyro angle.
162        return gyroValue;
163    }

213    /**
214     * Moves the robot a given distance at a given speed autonomously.
215     * 
216     * @param distance The distance in inches for the robot to travel.
217     * @param speed The speed the robot will travel.
218     */
219    public void move( double distance , double speed )
220    {
221        //Finds the amount of ticks the robot encoder needs to read before the robot stops.
222        distance *= TICKS_PER_INCH;
223        
224        //Starts the encoder reading.
225        _encoder.start();
226        
227        //Tells the robot to move forward at the given speed.
228        holoDrive( speed, 0.0, 0.0 );
229        
230        //While the encoder reading is less than the calculated distance...
231        while( _encoder.get() > distance )
232        {    
233            //Waits until the encoder reads the appropriate distance.
234            System.out.println( _encoder.get() );
235        }
236        
237        //After the robot has moved the appropriate distance, stop the robot.
238        stop();
239        
240        //Finally, stop and reset the encoder reading.
241        _encoder.stop();
242        _encoder.reset();
243    }

245    /**
246     * Turns the robot at a given angle.
247     * 
248     * @param angle The angle that the robot will turn in degrees.
249     */
250    public void turn( double angle, double speed )
251    {
252        //The flag which represents if the turn is done.
253        boolean done = false;
254        
255        //Finds the angle that the robot needs to turn to.
256        angle += Math.toDegrees( getGyro() );
257        
258        //Gets the three constants for the PID loop from the Smart Dashboard.
259        final double KP = SmartDashboard.getNumber( " P-Constant: " );
260        final double KI = SmartDashboard.getNumber( " I-Constant: " );
261        final double KD = SmartDashboard.getNumber( " D-Constant: " );
262        
263        //The error, the previous error, the sum of all errors, and the sum's limit that are used in the PID loop.
264        double error = 0.0;
265        double preError = 0.0;
266        double errorSum = 0.0;
267        final double sumLimit = 20.0;
268        
269        //The variables which represents the separate parts to the PID loop.
270        double p = 0.0;
271        double i = 0.0;
272        double d = 0.0;
273        
274        //The sum of the p, i, and d parts which is inputed into the motors.
275        double output = 0.0;
276        
277        //Create and start a timer for the time limit.
278        Timer timer = new Timer();
279        timer.start();
280        
281        //While the turn is not done...
282        while( !done )
283        {
284            //Sets the previous error to the error of the last interation of the loop.
285            preError = error;
286            
287            //Finds the current error between the given angle and the current angle of the robot.
288            error = ( angle - Math.toDegrees( getGyro() ) ) / 100;
289            error = Utility.limitRange( error );
290            
291            //Adds the new error to the error sum of the robot.
292            errorSum += error;
293            errorSum = Utility.limitRange( errorSum, sumLimit, -sumLimit );
294            
295            //Calculates the p, i, and d parts of the PID loop.
296            p = error * KP;
297            i = errorSum * KI;
298            d = ( preError - error ) * KD;
299            
300            //Adds the p, i, and d parts of the loop together, limits it between 1 and -1, then gives it to output.
301            output = Utility.limitRange( p + i + d, speed, -speed );
302            
303            //Turns the robot based on the output of the PID loop.
304            holoDrive( 0, 0, output );
305            
306            //If the error is below 5 percent or if it has been over 4 seconds, then end the loop.
307            if( Math.abs( error ) < 0.01 || timer.get() > 4.0 )
308            {
309                //Raise the flag to end the loop.
310                done = true;       
311            }
312        }
313        
314        //Stop the motors.
315        stop();
316    }

318    /**
319     * Line the Robot up with the wall.
320     * 
321     * @param speed The speed at which to square the robot.
322     */
323    public void square( double speed )
324    {
325        //Finds the original angle of the robot (which is the negative of the angle needed to turn).
326        double angle = Math.toDegrees( getGyro() );
327        
328        //If the angle is above 180...
329        if( angle >= 180 )
330        {
331            //Then change the angle to turn to turn more efficently.
332            angle = -( 360 - angle );
333        }
334        else if ( angle <= -180 )//Else if the angle is below -180...
335        {
336            //Then change the angle to turn to turn more efficently.
337            angle += 360;     
338        }
339        
340        //Turn the angle found to square the robot back to 0 degrees.
341        turn( -angle , speed ); 
342    }
