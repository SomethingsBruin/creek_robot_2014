022    private CCTalon _leftRear;
023    private CCTalon _rightRear;

036    private Chassis()

054        SmartDashboard.putNumber( " D-Constant: ", _KD );

075    /**
076     * Moves the robot using the mecanum wheels given a forward, slide, and
077     * rotation value.
078     * 
079     * @param fwd The forward value from a driver.
080     * @param sld The slide value from a driver.
081     * @param rot The rotation value from a driver.
082     */
083    public void holoDrive( double fwd, double sld, double rot )
084    {
085        //Calcuates the vector for each motor given the forward, slide, and rotation values.
086        double lf = fwd + sld + rot;
087        double rf = fwd - sld - rot;
088        double lr = fwd - sld + rot;
089        double rr = fwd + sld - rot;
090        
091        //Makes the max value equal to the left motor.
092        double maxVal = Math.abs( lf );
093        
094        //If any motor value is greater than the max value, make the max value equal to that motor value.
095        if ( Math.abs( rf ) > maxVal )
096        {
097            maxVal = Math.abs( rf );
098        }
099        
100        if ( Math.abs( lr ) > maxVal )
101        {
102            maxVal = Math.abs( lr );
103        }
104        
105        if ( Math.abs( rr ) > maxVal )
106        {
107            maxVal = Math.abs( rr );
108        }
109        
110        //If the max value was greater than one or less than negative one...
111        if( maxVal > 1.0 )
112        {
113            //Limit that motor value and scale the other motor values appropriatly.
114            lf /= maxVal;
115            rf /= maxVal;
116            lr /= maxVal;
117            rr /= maxVal;
118        }
119         
120        //Set each motor to its corresponding value.
121        _leftFront.set( lf );
122        _rightFront.set( rf );
123        _leftRear.set( lr );
124        _rightRear.set( rr );
125    }

127    /**
128     * Moves the robot based on the mecanum system given a forward value, a slide value,
129     * and a rotation value. Moves relative to driver instead of the starting position. 
130     * 
131     * @param fwd The forward value from a driver.
132     * @param sld The slide value from a driver.
133     * @param rot The rotation value from a driver.
134     */
135    public void relativeHoloDrive( double fwd, double sld, double rot )
136    {
137        //Calculates the relative forward and slide values.
138        double fwdTemp = fwd * Math.cos( getGyro() ) + sld * Math.sin( getGyro() );
139        double sldTemp = sld * Math.cos( getGyro() ) - fwd * Math.sin( getGyro() );
140        
141        //Inputs the calculated values into the normal holoDrive function.
142        holoDrive( fwdTemp, sldTemp, rot );     
143    }

145    /**
146     * Returns the angle of the gyro in radians on the chassis.
147     * 
148     * @return The angle of the gyro in radians.
149     */
150    public double getGyro()
151    {
152        //Gets the raw angle from the gyro
153        double gyroValue = _gyro.getAngle();
154        
155        //If the angle is greater than 360 or less than -360...
156        if ( Math.abs( gyroValue ) >= 360)
157        {
158            //Reset the gyro to zero degrees.
159            _gyro.reset();
160            gyroValue = _gyro.getAngle();
161        }
162        
163        //Converts the angle of the gyro from degrees to radians.
164        gyroValue = Math.toRadians( gyroValue );
165        
166        //Returns the gyro angle.
167        return gyroValue;
168    }

170    /**
171     * Moves the robot a given distance at a given speed autonomously.
172     * 
173     * @param distance The distance in inches for the robot to travel.
174     * @param speed The speed the robot will travel.
175     * 
176     * @return Returns whether the function moved or not.
177     */
178    public boolean move( double distance , double speed )
179    {
180        //Gets the inital distance from an object in front of the robot.
181        double initialDistance = _sonar.getDistance();
182        
183        //Chekcs if it can move the given distance, if not...
184        if( initialDistance - distance <= 0 )
185        {
186            //End the function and return false.
187            return false;
188        }
189        
190        //Makes sure that the speed is limited to 1 and -1.
191        speed = Utility.limitRange( speed );
192        
193        //While the distance traveled is less than distance wanted...
194        while( _sonar.getDistance() > initialDistance - distance )
195        {
196            //Drive the robot forward at the given speed.
197            holoDrive( speed , 0.0 ,0.0 );
198            System.out.println( initialDistance + " " + _sonar.getDistance() );
199        }
200        
201        //When the robot is done moving forward, stop the motors.
202        stop();
203        
204        //Return true if the robot moves.
205        return true;
206    }

208    /**
209     * Turns the robot at a given angle.
210     * 
211     * @param angle The angle that the robot will turn in degrees.
212     */
213    public void turn( double angle, double speed )
214    {
215        //The flag which represents if the turn is done.
216        boolean done = false;
217        
218        //Finds the angle that the robot needs to turn to.
219        angle += Math.toDegrees( getGyro() );
220        
221        //Gets the three constants for the PID loop from the Smart Dashboard.
222        final double KP = SmartDashboard.getNumber( " P-Constant: " );
223        final double KI = SmartDashboard.getNumber( " I-Constant: " );
224        final double KD = SmartDashboard.getNumber( " D-Constant: " );
225        
226        //The error, the previous error, the sum of all errors, and the sum's limit that are used in the PID loop.
227        double error = 0.0;
228        double preError = 0.0;
229        double errorSum = 0.0;
230        final double sumLimit = 20.0;
231        
232        //The variables which represents the separate parts to the PID loop.
233        double p = 0.0;
234        double i = 0.0;
235        double d = 0.0;
236        
237        //The sum of the p, i, and d parts which is inputed into the motors.
238        double output = 0.0;
239        
240        //While the turn is not done...
241        while( !done )
242        {
243            //Sets the previous error to the error of the last interation of the loop.
244            preError = error;
245            
246            //Finds the current error between the given angle and the current angle of the robot.
247            error = ( angle - Math.toDegrees( getGyro() ) ) / 100;
248            error = Utility.limitRange( error );
249            
250            //Adds the new error to the error sum of the robot.
251            errorSum += error;
252            errorSum = Utility.limitRange( errorSum, sumLimit, -sumLimit );
253            
254            //Calculates the p, i, and d parts of the PID loop.
255            p = error * KP;
256            i = errorSum * KI;
257            d = ( preError - error ) * KD;
258            
259            //Adds the p, i, and d parts of the loop together, limits it between 1 and -1, then gives it to output.
260            output = Utility.limitRange( p + i + d, speed, -speed );
261            
262            //Turns the robot based on the output of the PID loop.
263            holoDrive( 0, 0, output );
264            
265            //If the error is below 5 percent, then end the loop.
266            if( Math.abs( error ) < 0.01 )
267            {
268                //Raise the flag to end the loop.
269                done = true;       
270            }
271        }
272        
273        //Stop the motors.
274        stop();
275    }

277    /**
278     * Line the Robot up with the wall.
279     * 
280     * @param speed The speed at which to square the robot.
281     */
282    public void square( double speed )
283    {
284        //Finds the original angle of the robot (which is the negative of the angle needed to turn).
285        double angle = Math.toDegrees( getGyro() );
286        
287        //If the angle is above 180...
288        if( angle >= 180 )
289        {
290            //Then change the angle to turn to turn more efficently.
291            angle = -( 360 - angle );
292        }
293        else if ( angle <= -180 )//Else if the angle is below -180...
294        {
295            //Then change the angle to turn to turn more efficently.
296            angle += 360;     
297        }
298        
299        //Turn the angle found to square the robot back to 0 degrees.
300        turn( -angle , speed ); 
301    }

303    /**
304     * Stops all the motors on the chassis.
305     */
306    public void stop()
307    {
308        //Stops each motor individually.
309        _leftFront.stopMotor();
310        _leftRear.stopMotor();
311        _rightFront.stopMotor();
312        _rightRear.stopMotor();    
313    }
