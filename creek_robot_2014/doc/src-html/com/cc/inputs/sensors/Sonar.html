001package com.cc.inputs.sensors;
002
003import edu.wpi.first.wpilibj.AnalogChannel;
004
005/**
006 * A class which represents a sonar sensor.
007 */
008public class Sonar
009{
010    //The raw analog input.
011    private AnalogChannel _sonar;
012    
013    //The constant which represents how many ticks are in an inch for the sensor.
014    private final double _TICKSPERINCH = 4.722;
015    
016    /**
017     * A constructor which creates the sonar object. 
018     * 
019     * @param channel The channel the sonar is plugged into on the analog module.
020     */
021    public Sonar( int channel )
022    {
023        //Gets the analog input from the given channel.
024        _sonar = new AnalogChannel( channel ); 
025    }
026    
027    /**
028     * Gets the distance from an object in front of the sensor in inches.
029     * 
030     * @return Returns the distance from an object in inches.
031     */
032    public double getDistance()
033    {
034        //Finds the raw sonar value and converts it into inches.
035        double distance = _sonar.getValue();
036        distance /= _TICKSPERINCH;
037        
038        //Returns the inches from an object.
039        return distance;
040    }
041}
