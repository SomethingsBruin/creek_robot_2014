001package com.cc.inputs.sensors;
002
003import edu.wpi.first.wpilibj.ADXL345_I2C;
004
005/**
006 * The class which represents the acceleration of the robot in any given axis X, Y, or Z.
007 * Gives the acceleration of the axis in units of Gs.
008 */
009public class CCAccelerometer 
010{
011    //The accelerometer of the robot which gives us acceleration in either the X, Y, or Z axis.
012    private ADXL345_I2C _accelerometer;
013    
014    /**
015     * Initializes the accelerometer for the robot.
016     */
017    public CCAccelerometer()
018    {
019        //Initializes the accelerometer for digital module on 1 and for a range of -4 to 4 Gs.
020        _accelerometer = new ADXL345_I2C( 1, ADXL345_I2C.DataFormat_Range.k4G );
021    }
022    
023    /**
024     * Gets the acceleration of the robot on the x-axis relative to the robot.
025     * 
026     * @return The acceleration on the x-axis relative to the robot in Gs.
027     */
028    public double getXAccel()
029    {
030        //Finds the x-axis acceleration in Gs.
031        double xAccel = _accelerometer.getAcceleration( ADXL345_I2C.Axes.kY );
032        
033        //Returns the acceleration.
034        return xAccel;
035    }
036    
037    /**
038     * Gets the acceleration of the robot on the y-axis relative to the robot.
039     * 
040     * @return The acceleration on the y-axis relative to the robot in Gs.
041     */
042    public double getYAccel()
043    {
044        //Finds the x-axis acceleration in Gs and inverts the axis to be relative to the robot.
045        double yAccel = _accelerometer.getAcceleration( ADXL345_I2C.Axes.kX );
046        yAccel *= -1;
047        
048        //Returns the acceleration.
049        return yAccel;
050    }
051    
052    /**
053     * Gets the acceleration of the robot on the z-axis relative to the robot.
054     * 
055     * @return The acceleration on the z-axis relative to the robot in Gs.
056     */
057    public double getZAccel()
058    {
059        //Finds the z-axis acceleration in Gs.
060        double zAccel = _accelerometer.getAcceleration( ADXL345_I2C.Axes.kZ );
061        
062        //Returns the acceleration.
063        return zAccel;
064    }
065}
