001package com.cc.inputs.driver;
002
003import com.cc.utility.Utility;
004import edu.wpi.first.wpilibj.Joystick;
005
006/**
007 * The class which represents the driver type for the Airplane Controller.
008 */
009public class AirplaneController extends Driver
010{
011    //The contoller's X constants.
012    private final double _XMAX = 0.84252;
013    private final double _XMIN = -0.69531;
014    private final double _XCENTER = 0.0;
015    
016    //The contoller's Y constants. Y Max and Min are switched for normalization process.
017    private final double _YMAX = 0.65345;
018    private final double _YMIN = -0.57031;
019    private final double _YCENTER = 0.01575;
020    
021    //The contoller's Rotation constants.
022    private final double _ROTMAX = 0.84252;
023    private final double _ROTMIN = -0.72656;
024    private final double _ROTCENTER = 0.03149;
025    
026    //The contoller's Arm constants.
027    private final double _ARMMAX = -1.0;
028    private final double _ARMMIN = 1.0;
029    private final double _ARMCENTER = 0.0;
030    
031    //The joysticks of the controller.
032    private Joystick _joyOne;
033    private Joystick _joyTwo;
034    
035    /**
036     * Creates the airplane controller driver object.
037     */
038    private AirplaneController()
039    {
040        //Calls the contructor in Driver.
041        super();
042        
043        //Initializes the joysticks on channel 1 and 2.
044        _joyOne = new Joystick( 2 );
045        _joyTwo = new Joystick( 1 );
046    }
047    
048    /**
049     * Returns the singleton instance of the driver.
050     * 
051     * @return Returns the singleton instance of the driver.
052     */
053    public static Driver getInstance()
054    {
055        //If the dirver is not an AirplaneController, than make it an AirplaneController
056        if( !( _instance instanceof AirplaneController ) )
057        {
058            _instance = new AirplaneController();
059        }
060        
061        //Return the driver.
062        return _instance;
063    }
064    
065    /**
066     * Prints all the axes in the controller.
067     */
068    public void printAxes()
069    {
070        //Goes through 12 axis channels on the joystick and prints their current value.
071        for( int i = 0; i <= 12; i++ )
072        {
073            System.out.print( i + " One: " + _joyOne.getRawAxis( i ) + " " );
074            System.out.print( i + " Two: " + _joyTwo.getRawAxis( i ) + " " );
075        }
076        
077        //Moves to a new line.
078        System.out.println();
079    }
080  
081    /**
082     * Prints all the buttons in the controller.
083     */
084    public  void printButtons()
085    {
086        //Goes through 12 button channels on the joystick and prints their current value.
087        for( int i = 0; i <= 12; i++ )
088        {
089            System.out.print( i + " One: " + _joyOne.getRawButton( i ) + " " );
090            System.out.print( i + " Two: " + _joyTwo.getRawButton( i ) + " " );
091        }
092        
093        //Moves to a new line.
094        System.out.println();
095    }
096    
097    /**
098     * Gets the x value of the controller.
099     * 
100     * @return Returns the x value.
101     */
102    public  double getX()
103    {
104        //Finds the normalized x value of the controller, and then expos it.
105        double xValue = normalize( _joyOne.getRawAxis( 1 ), _XMIN, _XMAX, _XCENTER );
106        xValue = xValue * xValue * xValue;
107        
108        //Returns the x value.
109        return xValue;
110    }
111    
112    /**
113     * Gets the y value of the controller.
114     * 
115     * @return Returns the y value.
116     */
117    public double getY()
118    {
119        //Finds the normalized y value of the controller, and then expos it.
120        double yValue = -1 * normalize( _joyOne.getRawAxis( 2 ), _YMIN, _YMAX, _YCENTER );//Multiplies by -1 because the y-axis is inverted.
121        yValue = yValue * yValue * yValue;
122        
123        //Returns the y value.
124        return yValue;
125    }
126    
127    /**
128     * Gets the rotation value of the controller.
129     * 
130     * @return Returns the rotation value.
131     */
132    public double getRot()
133    {
134        //Finds the normalized rotation value of the controller, and then expos it.
135        double rValue = normalize( _joyOne.getRawAxis( 5 ), _ROTMIN, _ROTMAX, _ROTCENTER );
136        rValue = rValue * rValue * rValue;
137        
138        //Returns the rotation value.
139        return rValue;
140    }
141    
142    /**
143     * Gets the arm value of the controller.
144     * 
145     * @return Returns the arm value.
146     */
147    public double getArm()
148    {
149        //Finds the normalized arm value of the controller and limit range between 0.5 and -0.4.
150        double aValue = normalize( _joyTwo.getRawAxis( 3 ), _ARMMIN, _ARMMAX, _ARMCENTER );
151        aValue = Utility.limitRange( aValue, 0.5, -0.4 );
152        
153        //Returns the arm value.
154        return aValue;
155    }
156    
157    /**
158     * Gets the primary (Red) button state on the airplane controller.
159     * 
160     * @return Returns the primary button state.
161     */
162    public boolean getPriButton()
163    {
164        //Returns state of red button
165        boolean state = _joyOne.getRawButton( 1 );
166        return state;
167    }
168    
169    /**
170     * Gets the secondary (green) button state on the x-box controller.
171     * 
172     * @return Returns the secondary button state.
173     */
174    public boolean getSecButton()
175    {
176        //Returns state of the green button.
177        boolean state = _joyTwo.getRawButton( 1 );
178        return state;
179    }
180    
181    /**
182     * Gets the third (yellow) button state on the x-box controller.
183     * 
184     * @return Returns the third button state.
185     */
186    public boolean getThirdButton()
187    {
188        //Returns state of the yellow button.
189        boolean state = _joyTwo.getRawButton( 4 );
190        return state;
191    }
192
193    /**
194     * Gets the fourth (red) button state on the x-box controller.
195     * 
196     * @return Returns the fourth button state.
197     */
198    public boolean getFourthButton()
199    {
200        //Returns state of the red button.
201        boolean state = _joyTwo.getRawButton( 2 );
202        return state;
203    }
204
205    /**
206     * Gets the fifth (right) button state on the x-box controller.
207     * 
208     * @return Returns the fifth button state.
209     */
210    public boolean getFifthButton()
211    {
212        //Returns state of the right button.
213        boolean state = _joyTwo.getRawButton( 6 );
214        return state;
215    }
216
217    /**
218     * Gets the sixth (left) button state on the x-box controller.
219     * 
220     * @return Returns the sixth button state.
221     */
222    public boolean getSixthButton()
223    {
224        //Returns state of the left button.
225        boolean state = _joyTwo.getRawButton( 5 );
226        return state;
227    }
228}
