001package com.cc.inputs.driver;
002
003import com.cc.utility.Utility;
004
005import edu.wpi.first.wpilibj.Joystick;
006import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
007
008/**
009 * The class which represents the driver type for the Airplane Controller.
010 */
011public class AirplaneController extends Driver
012{
013    //The contoller's X constants.
014    private final double _XMAX = 0.84252;
015    private final double _XMIN = -0.69531;
016    private final double _XCENTER = 0.0;
017    
018    //The contoller's Y constants. Y Max and Min are switched for normalization process.
019    private final double _YMAX = 0.65345;
020    private final double _YMIN = -0.57031;
021    private final double _YCENTER = 0.01575;
022    
023    //The contoller's Rotation constants.
024    private final double _ROTMAX = 0.84252;
025    private final double _ROTMIN = -0.72656;
026    private final double _ROTCENTER = 0.03149;
027    
028    //The contoller's Arm constants.
029    private final double _ARMMAX = -1.0;
030    private final double _ARMMIN = 1.0;
031    private final double _ARMCENTER = 0.0;
032    
033    //The joysticks of the controller.
034    private Joystick _joyOne;
035    private Joystick _joyTwo;
036    
037    /**
038     * Creates the airplane controller driver object.
039     */
040    private AirplaneController()
041    {
042        //Calls the contructor in Driver.
043        super();
044        
045        //Initializes the joysticks on channel 1 and 2.
046        _joyOne = new Joystick( 2 );
047        _joyTwo = new Joystick( 1 );
048    }
049    
050    /**
051     * Returns the singleton instance of the driver.
052     * 
053     * @return Returns the singleton instance of the driver.
054     */
055    public static Driver getInstance()
056    {
057        //If the dirver is not an AirplaneController, than make it an AirplaneController
058        if( !( _instance instanceof AirplaneController ) )
059        {
060            _instance = new AirplaneController();
061        }
062        
063        //Return the driver.
064        return _instance;
065    }
066    
067    /**
068     * Prints all the axes in the controller.
069     */
070    public void printAxes()
071    {
072        //Goes through 12 axis channels on the joystick and prints their current value.
073        for( int i = 0; i <= 12; i++ )
074        {
075            System.out.print( i + " One: " + _joyOne.getRawAxis( i ) + " " );
076            System.out.print( i + " Two: " + _joyTwo.getRawAxis( i ) + " " );
077        }
078        
079        //Moves to a new line.
080        System.out.println();
081    }
082  
083    /**
084     * Prints all the buttons in the controller.
085     */
086    public  void printButtons()
087    {
088        //Goes through 12 button channels on the joystick and prints their current value.
089        for( int i = 0; i <= 12; i++ )
090        {
091            System.out.print( i + " One: " + _joyOne.getRawButton( i ) + " " );
092            System.out.print( i + " Two: " + _joyTwo.getRawButton( i ) + " " );
093        }
094        
095        //Moves to a new line.
096        System.out.println();
097    }
098    
099    /**
100     * Gets the x value of the controller.
101     * 
102     * @return Returns the x value.
103     */
104    public  double getX()
105    {
106        //Finds the normalized x value of the controller, and then expos it.
107        double xValue = normalize( _joyOne.getRawAxis( 1 ), _XMIN, _XMAX, _XCENTER );
108        xValue = Utility.expo( xValue, SmartDashboard.getNumber( " Driver Expo: " ) );
109        
110        //Returns the x value.
111        return xValue;
112    }
113    
114    /**
115     * Gets the y value of the controller.
116     * 
117     * @return Returns the y value.
118     */
119    public double getY()
120    {
121        //Finds the normalized y value of the controller, and then expos it.
122        double yValue = -1 * normalize( _joyOne.getRawAxis( 2 ), _YMIN, _YMAX, _YCENTER );//Multiplies by -1 because the y-axis is inverted.
123        yValue = Utility.expo( yValue, SmartDashboard.getNumber( " Driver Expo: " ) );
124        
125        //Returns the y value.
126        return yValue;
127    }
128    
129    /**
130     * Gets the rotation value of the controller.
131     * 
132     * @return Returns the rotation value.
133     */
134    public double getRot()
135    {
136        //Finds the normalized rotation value of the controller, and then expos it.
137        double rValue = normalize( _joyOne.getRawAxis( 5 ), _ROTMIN, _ROTMAX, _ROTCENTER );
138        rValue = Utility.expo( rValue, SmartDashboard.getNumber( " Driver Expo: " ) );
139        
140        //If going backwards...
141//        if( getY() < -SmartDashboard.getNumber( " Rot Dead-Zone: " ) )
142//        {
143//            //Then reverse the rotation value.
144//            rValue *= -1;
145//        }
146        
147        //Returns the rotation value.
148        return rValue;
149    }
150    
151    /**
152     * Gets the arm value of the controller.
153     * 
154     * @return Returns the arm value.
155     */
156    public double getArm()
157    {
158        //Finds the normalized arm value of the controller and limit range between 0.5 and -0.4.
159        double aValue = normalize( _joyTwo.getRawAxis( 3 ), _ARMMIN, _ARMMAX, _ARMCENTER );
160        aValue = Utility.limitRange( aValue, SmartDashboard.getNumber( " Arm Up-Speed: ") , SmartDashboard.getNumber( " Arm Down-Speed: " ) );
161        
162        //Returns the arm value.
163        return aValue;
164    }
165    
166    /**
167     * Gets the primary (Red) button state on the airplane controller.
168     * 
169     * @return Returns the primary button state.
170     */
171    public boolean getPriButton()
172    {
173        //Returns state of red button
174        boolean state = _joyOne.getRawButton( 1 );
175        return state;
176    }
177    
178    /**
179     * Gets the secondary (green) button state on the x-box controller.
180     * 
181     * @return Returns the secondary button state.
182     */
183    public boolean getSecButton()
184    {
185        //Returns state of the green button.
186        boolean state = _joyTwo.getRawButton( 1 );
187        return state;
188    }
189    
190    /**
191     * Gets the third (blue) button state on the x-box controller.
192     * 
193     * @return Returns the third button state.
194     */
195    public boolean getThirdButton()
196    {
197        //Returns state of the blue button.
198        boolean state = _joyTwo.getRawButton( 3 );
199        return state;
200    }
201
202    /**
203     * Gets the fourth (red) button state on the x-box controller.
204     * 
205     * @return Returns the fourth button state.
206     */
207    public boolean getFourthButton()
208    {
209        //Returns state of the red button.
210        boolean state = _joyTwo.getRawButton( 2 );
211        return state;
212    }
213    
214    /**
215     * Gets the fifth (yellow) button state on the x-box controller.
216     * 
217     * @return Returns the fifth the button state.
218     */
219    public boolean getFifthButton()
220    {
221        //Returns state of the yellow.
222        boolean state = _joyTwo.getRawButton( 4 );
223        return state;
224    }
225
226    /**
227     * Gets the sixth (right) button state on the x-box controller.
228     * 
229     * @return Returns the sixth button state.
230     */
231    public boolean getSixthButton()
232    {
233        //Returns state of the right button.
234        boolean state = _joyTwo.getRawButton( 6 );
235        return state;
236    }
237
238    /**
239     * Gets the seventh (left) button state on the x-box controller.
240     * 
241     * @return Returns the seventh button state.
242     */
243    public boolean getSeventhButton()
244    {
245        //Returns state of the left button.
246        boolean state = _joyTwo.getRawButton( 5 );
247        return state;
248    }
249}
