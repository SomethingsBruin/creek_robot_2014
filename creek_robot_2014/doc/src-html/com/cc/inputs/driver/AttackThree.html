001package com.cc.inputs.driver;
002
003import edu.wpi.first.wpilibj.Joystick;
004
005/**
006 * The class which represents the driver type for the XBox Controller.
007 */
008public class AttackThree extends Driver
009{
010    //The contoller's X constants.
011    private final double _XMAX = 1.0;
012    private final double _XMIN = -1.0;
013    private final double _XCENTER = 0.0;
014    
015    //The contoller's Y constants. Y Max and Min are switched for normalization process.
016    private final double _YMAX = 1.0;
017    private final double _YMIN = -1.0;
018    private final double _YCENTER = 0.0;
019    
020    //The contoller's Rotation constants.
021    private final double _ROTMAX = 1.0;
022    private final double _ROTMIN = -1.0;
023    private final double _ROTCENTER = 0.0;
024
025    //The joysticks of the controller.
026    private Joystick _attackOne; 
027    private Joystick _attackTwo;
028    
029    /**
030     * Creates the joystick driver object.
031     */
032    private AttackThree()
033    {
034        //Calls the contructor in Dirver.
035        super();
036        
037        //Initializes the joystick on channel 1 and channel 2.
038        _attackOne = new Joystick( 1 );
039        _attackTwo = new Joystick( 2 );
040    }
041    
042    /**
043     * Returns the singleton instance of the driver.
044     * 
045     * @return Returns the singleton instance of the driver.
046     */
047    public static Driver getInstance()
048    {
049        //If the dirver is not an AttackThree, than make it an AttackThree
050        if( !( _instance instanceof AttackThree ) )
051        {
052            _instance = new AttackThree();
053        }
054        
055        //Return the driver.
056        return _instance;
057    }
058    
059    /**
060     * Prints all the axes in the controller.
061     */
062    public void printAxes()
063    {
064        //Goes through 12 axis channels on the joystick and prints their current value.
065        for( int i = 0; i <= 12; i++ )
066        {
067            System.out.print( i + "-One: " + _attackOne.getRawAxis( i ) + " " );
068            System.out.print( i + "-Two: " + _attackTwo.getRawAxis( i ) + " " );
069        }
070        
071        //Moves to a new line.
072        System.out.println();
073    }
074  
075    /**
076     * Prints all the buttons in the controller.
077     */
078    public  void printButtons()
079    {
080        //Goes through 12 button channels on the joystick and prints their current value.
081        for( int i = 0; i <= 12; i++ )
082        {
083            System.out.print( i + "-One: " + _attackOne.getRawButton( i ) + " " );
084            System.out.print( i + "-Two: " + _attackTwo.getRawButton( i ) + " " );
085        }
086        
087        //Moves to a new line.
088        System.out.println();
089    }
090    
091    /**
092     * Gets the x value of the controller.
093     * 
094     * @return Returns the x value.
095     */
096    public  double getX()
097    {
098        //Finds the normalized x value of the controller, and then expos it.
099        double xValue = normalize( _attackOne.getRawAxis( 1 ), _XMIN, _XMAX, _XCENTER );
100        xValue = xValue * xValue * xValue;
101        
102        //Returns the x value.
103        return xValue;
104    }
105    
106    /**
107     * Gets the y value of the controller.
108     * 
109     * @return Returns the y value.
110     */
111    public double getY()
112    {
113        //Finds the normalized y value of the controller, and then expos it.
114        double yValue = -1 * normalize( _attackOne.getRawAxis( 2 ), _YMIN, _YMAX, _YCENTER );//Multiplies by -1 because the y-axis is inverted.
115        yValue = yValue * yValue * yValue;
116        
117        //Returns the y value.
118        return yValue;
119    }
120    
121    /**
122     * Gets the rotation value of the controller.
123     * 
124     * Returns the rotation value.
125     */
126    public double getRot()
127    {
128        //Finds the normalized rotation value of the controller, and then expos it.
129        double rValue = normalize( _attackTwo.getRawAxis( 1 ), _ROTMIN, _ROTMAX, _ROTCENTER );
130        rValue = rValue * rValue * rValue;
131        
132        //Returns the rotation value.
133        return rValue;
134    }
135    
136    /**
137     * Gets the primary (Joystick One Trigger) button state.
138     * 
139     * @return Returns the primary button state.
140     */
141    public boolean getPriButton()
142    {
143        //Returns state of Joystick One Trigger.
144        boolean state = _attackOne.getRawButton( 1 );
145        return state;
146    }
147    
148    /**
149     * Gets the secondary (Joystick Two Trigger) button state.
150     * 
151     * @return Returns the secondary button state.
152     */
153    public boolean getSecButton()
154    {
155        //Returns state of Joystick Two Trigger.
156        boolean state = _attackTwo.getRawButton( 1 );
157        return state;
158    }
159    
160    /**
161     * Gets the third (green) button state.
162     * 
163     * @return Returns the third button state.
164     */
165    public boolean getThirdButton()
166    {
167        //Returns state of the third button.
168        boolean state = _attackOne.getRawButton( 2 );
169        return state;
170    }
171
172    /**
173     * Gets the fourth (red) button state.
174     * 
175     * @return Returns the fourth button state.
176     */
177    public boolean getFourthButton()
178    {
179        //Returns state of the fourth button.
180        boolean state = _attackTwo.getRawButton( 2 );
181        return state;
182    }
183
184    /**
185     * Gets the fifth button state.
186     * 
187     * @return Returns the fifth button state.
188     */
189    public boolean getFifthButton()
190    {
191        //Returns state of the fifth button.
192        boolean state = _attackOne.getRawButton( 3 );
193        return state;
194    }
195
196    /**
197     * Gets the sixth button state.
198     * 
199     * @return Returns the sixth button state.
200     */
201    public boolean getSixthButton()
202    {
203        //Returns state of the sixth button.
204        boolean state = _attackTwo.getRawButton( 3 );
205        return state;
206    }
207}
