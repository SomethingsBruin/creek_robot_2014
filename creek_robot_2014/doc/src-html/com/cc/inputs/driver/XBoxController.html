001package com.cc.inputs.driver;
002
003import com.cc.utility.Utility;
004
005import edu.wpi.first.wpilibj.Joystick;
006
007/**
008 * Class representing XBox Controller.
009 */
010public class XBoxController extends Driver
011{
012   //The contoller's X constants.
013    private final double _XMAX = 1.0;
014    private final double _XMIN = -1.0;
015    private final double _XCENTER = 0.0;
016    
017    //The contoller's Y constants. Y Max and Min are switched for normalization process.
018    private final double _YMAX = 1.0;
019    private final double _YMIN = -1.0;
020    private final double _YCENTER = -0.0078125;
021    
022    //The contoller's Rotation constants.
023    private final double _ROTMAX = 1.0;
024    private final double _ROTMIN = -1.0;
025    private final double _ROTCENTER = 0.0;
026    
027    //The joystick of the controller.
028    private Joystick _joy;
029    
030    /**
031     * Creates the XBox Controller driver object.
032     */
033    private XBoxController()
034    {
035        //Calls the contructor in Dirver.
036        super();
037        
038        //Initializes the joysticks on channel 1 and 2.
039        _joy = new Joystick( 1 );
040    }
041    
042    /**
043     * Returns the singleton instance of the driver.
044     * 
045     * @return Returns the singleton instance of the driver.
046     */
047    public static Driver getInstance()
048    {
049        //If the dirver is not an XBox Controller, than make it an XBoxController
050        if( !( _instance instanceof XBoxController ) )
051        {
052            _instance = new XBoxController();
053        }
054        
055        //Return the driver.
056        return _instance;
057    }
058    
059    /**
060     * Prints all the axes in the controller.
061     */
062    public void printAxes()
063    {
064        //Goes through 12 axis channels on the joystick and prints their current value.
065        for( int i = 0; i <= 12; i++ )
066        {
067            System.out.print( i + " One: " + _joy.getRawAxis( i ) + " " );
068        }
069        
070        //Moves to a new line.
071        System.out.println();
072    }
073  
074    /**
075     * Prints all the buttons in the controller.
076     */
077    public  void printButtons()
078    {
079        //Goes through 12 button channels on the joystick and prints their current value.
080        for( int i = 0; i <= 12; i++ )
081        {
082            System.out.print( i + " One: " + _joy.getRawButton( i ) + " " );
083        }
084        
085        //Moves to a new line.
086        System.out.println();
087    }
088    
089    /**
090     * Gets the x value of the controller.
091     * 
092     * @return Returns the x value.
093     */
094    public  double getX()
095    {
096        //Finds the normalized x value of the controller, and then expos it.
097        double xValue = normalize( _joy.getRawAxis( 4 ), _XMIN, _XMAX, _XCENTER );
098      //  xValue = xValue * xValue * xValue;
099        
100        //Returns the x value.
101        return xValue;
102    }
103    
104    /**
105     * Gets the y value of the controller.
106     * 
107     * @return Returns the y value.
108     */
109    public double getY()
110    {
111        //Finds the normalized y value of the controller, and then expos it.
112        double yValue = -1 * normalize( _joy.getRawAxis( 5 ), _YMIN, _YMAX, _YCENTER );//Multiplies by -1 because the y-axis is inverted.
113        // yValue = yValue * yValue * yValue;
114        
115        //Returns the y value.
116        return yValue;
117    }
118    
119    /**
120     * Gets the rotation value of the controller.
121     * 
122     * Returns the rotation value.
123     */
124    public double getRot()
125    {
126        //Finds the normalized rotation value of the controller, and then expos it.
127        double rValue = normalize( _joy.getRawAxis( 1 ), _ROTMIN, _ROTMAX, _ROTCENTER );
128        //rValue = rValue * rValue * rValue;
129        
130        //Limits the turn of the robot with this controller to only half.
131        rValue = Utility.limitRange( rValue, 0.5, -0.5 );
132        
133        //Returns the rotation value.
134        return rValue;
135    }
136    
137    /**
138     * Gets the primary (Right Trigger) button state.
139     * 
140     * @return Returns the primary button state.
141     */
142    public boolean getPriButton()
143    {
144        //Returns state of right trigger.
145        boolean state = _joy.getRawButton( 6 );
146        return state;
147    }
148    
149    /**
150     * Gets the secondary (Left Trigger) button state.
151     * 
152     * @return Returns the secondary button state.
153     */
154    public boolean getSecButton()
155    {
156        //Returns state of left trigger.
157        boolean state = _joy.getRawButton( 5 );
158        return state;
159    }     
160
161    /**
162     * Gets the third (green) button state.
163     * 
164     * @return Returns the third button state.
165     */
166    public boolean getThirdButton()
167    {
168        //Returns state of the green button.
169        boolean state = _joy.getRawButton( 1 );
170        return state;
171    }
172
173    /**
174     * Gets the fourth (red) button state.
175     * 
176     * @return Returns the fourth button state.
177     */
178    public boolean getFourthButton()
179    {
180        //Returns state of the red button.
181        boolean state = _joy.getRawButton( 2 );
182        return state;
183    }
184
185    /**
186     * Gets the fifth (blue) button state.
187     * 
188     * @return Returns the fifth button state.
189     */
190    public boolean getFifthButton()
191    {
192        //Returns state of the blue button.
193        boolean state = _joy.getRawButton( 3 );
194        return state;
195    }
196
197    /**
198     * Gets the sixth (yellow) button state.
199     * 
200     * @return Returns the sixth button state.
201     */
202    public boolean getSixthButton()
203    {
204        //Returns state of the yellow button.
205        boolean state = _joy.getRawButton( 4 );
206        return state;
207    }
208}
