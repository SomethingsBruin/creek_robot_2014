001package com.cc.utility;
002
003import com.sun.squawk.util.MathUtils;
004
005/**
006 * This class is used for all math that limits a number between to numbers.
007 * The most common instance of this is when you are limiting a number between
008 * 1 and -1.
009 */
010public class Utility
011{
012    /**
013     * Limits a number between 1.0 and -1.0.
014     * 
015     * @param val The value you wish to limit between 1.0 and -1.0.
016     * 
017     * @return Returns the limited number.
018     */
019    public static double limitRange( double val )
020    {
021        if( val > 1.0 )//If number is greater than 1.0, it becomes 1.0.
022        {
023            val = 1.0;  
024        }
025        else if( val < -1.0 )//Else if number is less than -1.0, it becomes -1.0.
026        {
027            val = -1.0;
028        }
029        
030        //Returns the limited number
031        return val;
032    }
033    
034    /**
035     * Limits a number between a given maximum and a given minimum.
036     * 
037     * @param val The value you wish to limit between a max and min.
038     * @param max The max to limit the value on.
039     * @param min The min to limit the value on.
040     * 
041     * @return Returns the limited number.
042     */
043    public static double limitRange( double val , double max , double min )
044    {
045        //If number is greater than maximum, it becomes the maximum.
046        if( val > max )
047        {
048            val = max;  
049        }
050        else if( val < min )//Else if the number is less than minimum, it becomes the minimum.
051        {
052            val = min;
053        }
054        
055        //Returns the limited number.
056        return val;
057    }
058    
059    /**
060     * Returns a number that is equal to the given value to the power of the given exponent.
061     * If the number is negative going into the method, it will still be negative coming 
062     * out of the method.
063     * 
064     * @param value The value that will be put to the power of the given exponent.
065     * @param expo The exponent that the value is to the power to.
066     * 
067     * @return Returns the value to the power of the given exponent.
068     */
069    public static double expo( double value, double expo )
070    {
071        //A boolean which represents whether the original value was negative or not.
072        boolean negative = value < 0.0;
073        
074        //Does the math for finding the new value.
075        value = MathUtils.pow( value, expo );
076        
077        //If the value was originally negative and the value is now positive...
078        if( negative && value > 0.0 )
079        {
080            //Make the value negative.
081            value *= -1;
082        }
083        
084        //Return the new value.
085        return value;
086    }
087}
