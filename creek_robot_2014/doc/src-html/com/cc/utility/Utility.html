001package com.cc.utility;
002
003/**
004 * This class is used for all math that limits a number between to numbers.
005 * The most common instance of this is when you are limiting a number between
006 * 1 and -1.
007 */
008public class Utility
009{
010    /**
011     * Limits a number between 1.0 and -1.0.
012     * 
013     * @param val The value you wish to limit between 1.0 and -1.0.
014     * 
015     * @return Returns the limited number.
016     */
017    public static double limitRange( double val )
018    {
019        if( val > 1.0 )//If number is greater than 1.0, it becomes 1.0.
020        {
021            val = 1.0;  
022        }
023        else if( val < -1.0 )//Else if number is less than -1.0, it becomes -1.0.
024        {
025            val = -1.0;
026        }
027        
028        //Returns the limited number
029        return val;
030    }
031    
032    /**
033     * Limits a number between a given maximum and a given minimum.
034     * 
035     * @param val The value you wish to limit between a max and min.
036     * @param max The max to limit the value on.
037     * @param min The min to limit the value on.
038     * 
039     * @return Returns the limited number.
040     */
041    public static double limitRange( double val , double max , double min )
042    {
043        //If number is greater than maximum, it becomes the maximum.
044        if( val > max )
045        {
046            val = max;  
047        }
048        else if( val < min )//Else if the number is less than minimum, it becomes the minimum.
049        {
050            val = min;
051        }
052        
053        //Returns the limited number.
054        return val;
055    } 
056}
