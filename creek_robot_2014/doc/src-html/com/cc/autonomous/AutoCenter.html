001package com.cc.autonomous;
002
003import edu.wpi.first.wpilibj.Timer;
004
005/**
006 * The class which represents the AutoCommand which starts in the center position
007 * figures out which goal is the hot goal, moves forward into position, turns towards
008 * the hot goal, and then shoots the ball.
009 */
010public class AutoCenter extends AutoCommand
011{
012    /**
013     * Calls the constructor from AutoCommand.
014     */
015    public AutoCenter()
016    {
017        super();
018    }
019    
020    /**
021     * Runs this particular AutoCommand which starts in the center position
022     * figures out which goal is the hot goal, moves forward into position, turns towards
023     * the hot goal, and then shoots the ball.
024     */
025    public void runAutoCommand()
026    {
027        //Moves the chassis forward 12 feet at 0.75 speed.
028        _chassis.move( 144, 0.75 );
029        
030        //Tunrs the robot 20 degrees at 0.75 speed.
031        _chassis.turn( 20, 0.75 );
032        
033        //Set the arm to the correct orientation.
034        setArm();
035        
036        //Shoots the mechanism.
037        _mechanism.shoot();
038        
039        // Delays the program. 
040        Timer.delay( 1.0 );
041        
042        //Turns the robot back 20 degrees at 0.75 speed.
043        _chassis.square( 0.75 );
044    }
045}
