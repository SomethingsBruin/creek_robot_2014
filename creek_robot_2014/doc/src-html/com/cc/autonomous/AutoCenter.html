001package com.cc.autonomous;
002
003/**
004 * The class which represents the AutoCommand which starts in the center position
005 * figures out which goal is the hot goal, moves forward into position, turns towards
006 * the hot goal, and then shoots the ball.
007 */
008public class AutoCenter extends AutoCommand
009{
010    /**
011     * Calls the constructor from AutoCommand.
012     */
013    public AutoCenter()
014    {
015        super();
016    }
017    
018    /**
019     * Runs this particular AutoCommand which starts in the center position
020     * figures out which goal is the hot goal, moves forward into position, turns towards
021     * the hot goal, and then shoots the ball.
022     */
023    public void runAutoCommand()
024    {
025        //Moves the chassis forward 12 feet at 0.75 degrees.
026        _chassis.move( 144, 0.75 );
027        
028        //Tunrs the robot 20 degrees at 0.75 degrees.
029        _chassis.turn( 20, 0.75 );
030        
031        //Shoots the mechanism.
032        
033        //Turns the robot back 20 degrees at 0.75 degrees.
034        _chassis.square( 0.75 );
035    }
036}
