001package com.cc.autonomous;
002
003import com.cc.systems.Chassis;
004import com.cc.systems.Mechanism;
005import edu.wpi.first.wpilibj.Timer;
006
007/**
008 * The abstract class which represents any command that is called for Autonomous
009 * but does not define any particular auto command.
010 */
011public abstract class AutoCommand 
012{
013    //The chassis that is used in autonomous.
014    protected Chassis _chassis;
015    
016    //The mechanism that is used in autonomous.
017    protected Mechanism _mechanism;
018    
019    /**
020     * Default constructor for an AutoCommand that will get the singleton objects
021     * of the chassis and the mechanism.
022     */
023    public AutoCommand()
024    {
025        //Gets the singleton of the chassis and the mechanism.
026        _chassis = Chassis.getInstance();
027        _mechanism = Mechanism.getInstance();
028    }
029    
030    /**
031     * The method which runs the given AutoCommand.
032     */
033    public abstract void runAutoCommand();
034    
035    /**
036     * Raises the arm to the correct orientation to shoot in autonomous.
037     */
038    protected void setArm()
039    {
040        //Raise the arm at 0.5 speed for 1.5 seconds.
041        _mechanism.raiseArm( 0.5 );      
042        Timer.delay( 1.5 );
043        
044        //Stop raising the arm.
045        _mechanism.stopArm();
046    }
047}
