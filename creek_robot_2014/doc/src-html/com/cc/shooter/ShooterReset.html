001package com.cc.shooter;
002
003import com.cc.systems.Mechanism;
004import edu.wpi.first.wpilibj.Timer;
005
006/**
007 * The class which is a thread that shoots the shooter then resets the shooter.
008 */
009public class ShooterReset extends Thread
010{
011    //The shooter object in the thread.
012    Shooter _shooter;
013    
014    //The mechanism object in the thread.
015    Mechanism _mechanism;
016    
017    /**
018     * Creates the thread which shoots the shooter and then resets the shooter.
019     * 
020     * @param shooter The object which the thread will shoot and reset.
021     */
022    public ShooterReset( Shooter shooter )
023    {
024        //Initializes the shooter and mechanism object in the class.
025        _shooter = shooter;
026        _mechanism = Mechanism.getInstance();
027    }
028    
029    public void run()
030    {
031        //Start ejecting the ball and wait one second to shoot the ball.
032        _mechanism.eject();
033        
034        //Wait 0.2 seconds to shoot the mechanism.
035        Timer.delay( 0.1 );
036        
037        //Run the shooter until limit switch is released
038        _shooter.turnOn();
039
040        //Wait one second so it can clear the switch before monitoring the limit switch again.
041        Timer.delay( 1.0 );
042        
043        //Turn off the ejection.
044        _mechanism.stopIntake();
045        
046        //Run motor until limit switch is pressed
047        while ( _shooter.getLimit() == false )
048        {
049            //Do nothing.
050            ShooterReset.yield();
051        }
052        
053        //Turn off the shooter when the limit switch is pressed.
054        _shooter.turnOff();
055    }
056}
